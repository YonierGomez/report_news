🚀 Inicio rápido

docker run -d --name report_news \
  -e TOKEN="tu_token_telegram" \
  neytor/report_news
# O con docker-compose
version: '3'
services:
  report_news:
    image: neytor/report_news
    restart: always
    environment:
      - TOKEN=tu_token_telegram
